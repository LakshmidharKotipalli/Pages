Tree 1: LangChain Fundamentals & Architecture

Interviewer:

What is LangChain and why would you use it instead of calling OpenAI's API directly?

Follow-up 1:

Follow-up 2:

Candidate:

Initial Answer: LangChain is an orchestration framework for building applications with LLMs. While direct API calls work for simple use cases, LangChain provides structured components for complex workflows.

Key advantages:

  • Abstraction over different LLM providers (OpenAI, Anthropic, local models)
  • Prompt template management with versioning
  • Memory management for multi-turn conversations
  • Chain composition for sequential processing
  • Integration with 50+ tools (databases, APIs, file systems)

)

# With LangChain - structured, composable

from langchain.chains import LLMChain

result = chain.invoke({"topic": "quantum computing"})

Follow-up 2 - Chain vs LangGraph:

Chains: Linear, deterministic execution flow. Good for fixed pipelines like: input → retrieve → prompt → LLM → output

LangGraph Agents: Cyclic, stateful graphs with decision loops. The agent decides what to do next based on current state and available tools.

# Chain - linear
rag_chain = retriever | prompt | llm | parser

# LangGraph Agent - cyclic with routing
from langgraph.graph import StateGraph, END
from typing import TypedDict, Annotated

class AgentState(TypedDict):
    messages: Annotated[list, operator.add]
    next_action: str

def should_continue(state: AgentState):
    if state["next_action"] == "end":
        return END
    return "continue"

workflow = StateGraph(AgentState)
workflow.add_node("agent", agent_node)
workflow.add_node("tools", tool_node)
workflow.add_conditional_edges(
    "agent",
    should_continue,
    {"continue": "tools", "end": END}
)
workflow.add_edge("tools", "agent")
app = workflow.compile()

Key Takeaway:

Use Chains for predictable, linear workflows. Use LangGraph when you need an agent that makes decisions, loops, and dynamically routes between tools based on intermediate results.

{context}

Answer:"""

Initial Answer - Agent Architecture:

An Agent is a system that uses an LLM as a reasoning engine to decide which actions to take. The loop is:

  1. LLM receives user query + available tools
  2. LLM decides which tool to use (or answer directly)
  3. Tool executes and returns observation
  4. LLM receives observation and decides next action
  5. Repeat until LLM decides it has the final answer
from langchain.agents import create_openai_functions_agent, AgentExecutor
from langchain_openai import ChatOpenAI
from langchain.tools import tool

# Define custom tools
@tool

    """Search internal documentation."""
    # Implementation
    return "Relevant docs content..."

@tool
def query_database(sql: str) -> str:
    """Execute SQL query on database."""
    # Implementation
    return "[{name: 'Alice', age: 30}]"

tools = [search_documents, query_database]

# Create agent

# Create executor with error handling

    agent=agent,

import operator

def router(state: AgentState) -> str:

    if hasattr(last_message, "tool_calls"):
        return "tools"
    return END

    response = llm.invoke(state["messages"])
    return {"messages": [response]}

    # Execute tool calls
    pass

# Build graph
graph = StateGraph(AgentState)
graph.add_node("agent", agent_node)
graph.add_node("tools", tool_node)
graph.set_entry_point("agent")
graph.add_conditional_edges("agent", router)

import requests

@tool
def get_weather(city: str) -> str:
    """Get current weather for a city.
    Args:
        city: Name of the city (e.g., 'San Francisco, CA')

    response = requests.get(url).json()
    return f"{response['current']['temp_f']}°F, {response['current']['condition']['text']}"

# Method 2: StructuredTool with validation
class CalculatorInput(BaseModel):
    a: float = Field(description="First number")
    b: float = Field(description="Second number")
    operation: str = Field(description="Operation: add, subtract, multiply, divide")

def calculator(a: float, b: float, operation: str) -> str:
    operations = {
        "add": a + b,
        "subtract": a - b,
        "multiply": a * b,
        "divide": a / b if b != 0 else "Error: Division by zero"
    }
    result = operations.get(operation, "Invalid operation")
    return str(result)

calculator_tool = StructuredTool.from_function(
    func=calculator,
    name="calculator",
    description="Perform basic arithmetic operations",
    args_schema=CalculatorInput
)

Production Architecture:

# Multi-tier memory strategy
import json
from datetime import datetime, timedelta

class ProductionMemoryManager:
    def __init__(self, redis_client, postgres_pool):
        self.redis = redis_client  # Hot storage (recent)
        self.postgres = postgres_pool  # Cold storage (archive)

    async def get_conversation(self, user_id: str, session_id: str):
        key = f"conversation:{user_id}:{session_id}"

        # Try Redis cache first (hot)
        cached = await self.redis.get(key)
        if cached:
            return json.loads(cached)

        # Fallback to PostgreSQL (cold)
        query = """
            SELECT messages FROM conversations
            WHERE user_id = $1 AND session_id = $2
            ORDER BY timestamp DESC LIMIT 100
        """
        rows = await self.postgres.fetch(query, user_id, session_id)

        messages = [row["messages"] for row in rows]

        # Cache in Redis
        await self.redis.setex(
            key,
            3600,  # 1 hour TTL
            json.dumps(messages)
        )

        return messages

    async def add_message(self, user_id: str, session_id: str, role: str, content: str):
        message = {
            "role": role,
            "content": content,
            "timestamp": datetime.utcnow().isoformat()
        }

        # Write to Redis
        key = f"conversation:{user_id}:{session_id}"
        await self.redis.lpush(key, json.dumps(message))

        # Async write to PostgreSQL
        await self.postgres.execute(
            "INSERT INTO messages (user_id, session_id, role, content, timestamp) VALUES ($1, $2, $3, $4, $5)",
            user_id, session_id, role, content, datetime.utcnow()
        )

from langchain.memory import VectorStoreMemoryWrapper

class UserProfileMemory:
    def __init__(self, vector_store):
        self.vsm = VectorStoreMemoryWrapper(
            vectorstore=vector_store,
            memory_key="user_profile",
            input_key="input",
            output_key="output"

    async def remember(self, user_id: str, fact: str):
        """Store important facts about user."""

import time
import random
from functools import wraps

def exponential_backoff(
    max_retries=5,
    initial_delay=1,
    max_delay=60,
    exponential_base=2,
    jitter=True
):
    """
    Exponential backoff with jitter for distributed systems.

    Retry delays: 1s, 2s, 4s, 8s, 16s (with jitter)
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            delay = initial_delay

            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_retries - 1:
                        logger.error(f"All {max_retries} retries failed")
                        raise

                    # Calculate delay with exponential backoff
                    current_delay = min(delay, max_delay)

                    # Add jitter to avoid thundering herd
                    if jitter:
                        current_delay = random.uniform(0, current_delay)

                    logger.warning(
                        f"Attempt {attempt + 1} failed: {e}. "
                        f"Retrying in {current_delay:.2f}s"
                    )

                    time.sleep(current_delay)

                    # Exponential increase
                    delay *= exponential_base

        return wrapper
    return decorator

# Circuit breaker implementation
from dataclasses import dataclass
from datetime import datetime, timedelta
from enum import Enum

class CircuitState(Enum):
    CLOSED = "closed"      # Normal operation
    OPEN = "open"          # Failing, reject requests
    HALF_OPEN = "half_open"  # Testing recovery

@dataclass
class CircuitBreaker:
    failure_threshold: int = 5
    recovery_timeout: int = 60  # seconds
    failure_count: int = 0
    last_failure_time: datetime = None
    state: CircuitState = CircuitState.CLOSED

    def record_failure(self):
        self.failure_count += 1
        self.last_failure_time = datetime.now()

        if self.failure_count >= self.failure_threshold:
            self.state = CircuitState.OPEN
            logger.error("Circuit breaker opened")

    def record_success(self):
        self.failure_count = 0
        if self.state == CircuitState.HALF_OPEN:
            self.state = CircuitState.CLOSED
            logger.info("Circuit breaker recovered")

    def can_attempt(self) -> bool:
        if self.state == CircuitState.CLOSED:
            return True

        if self.state == CircuitState.OPEN:
            if datetime.now() - self.last_failure_time > timedelta(seconds=self.recovery_timeout):
                self.state = CircuitState.HALF_OPEN
                logger.info("Circuit breaker half-open, testing recovery")
                return True
            return False

        return True  # HALF_OPEN

# Usage in chain
circuit_breaker = CircuitBreaker()

def call_with_circuit_breaker(func):
    def wrapper(*args, **kwargs):
        if not circuit_breaker.can_attempt():
            raise Exception("Circuit breaker is OPEN, using fallback")

        try:
            result = func(*args, **kwargs)
            circuit_breaker.record_success()
            return result
        except Exception as e:
            circuit_breaker.record_failure()
            raise e
    return wrapper

    'Total requests',
    ['endpoint', 'status']

    'langchain_request_duration_seconds',
    'Request duration',
    ['endpoint']
)

token_usage = Counter(
    'langchain_tokens_total',
    'Token usage',
    ['model', 'type']  # type: prompt/completion
)

    'langchain_active_sessions',
    'Active user sessions'
)

# Middleware to track metrics
@app.middleware("http")
async def metrics_middleware(request, call_next):
    start = time.time()

    response = await call_next(request)

    # Record metrics
    duration = time.time() - start
    request_count.labels(
        endpoint=request.url.path,
        status=response.status_code
    ).inc()

    request_duration.labels(
        endpoint=request.url.path
    ).observe(duration)

    return response

# Expose metrics endpoint
metrics_app = make_asgi_app()

# Custom tracing
from langchain.callbacks import BaseCallbackHandler

class MetricsCallback(BaseCallbackHandler):
    def __init__(self):
        self.llm_calls = 0
        self.total_tokens = 0

    def on_llm_end(self, response, **kwargs):
        self.llm_calls += 1
        self.total_tokens += response.llm_output.get("token_usage", {}).get("total_tokens", 0)

        # Log to Prometheus
        token_usage.labels(
            model=response.llm_output.get("model", "unknown"),
            type="total"
        ).inc(self.total_tokens)
