2.1 Retry Strategies with Exponential Backoff

import time
import random
from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
    before_sleep_log
)
import logging

logger = logging.getLogger(__name__)

class LLMRetryStrategy:
    """
    Production-grade retry strategy for LLM calls.

    Handles:
    - Rate limiting (429)
    - Connection errors
    - Timeout errors
    - Server errors (5xx)
    """

    @staticmethod
    @retry(
        stop=stop_after_attempt(5),  # Max 5 attempts
        wait=wait_exponential(
            multiplier=1,      # Start with 1s delay
            min=1,            # Minimum wait
            max=60            # Maximum wait
        ),
        retry=retry_if_exception_type((
            openai.RateLimitError,
            openai.APIConnectionError,
            openai.APITimeoutError,
            openai.APIError  # Retry on 5xx errors
        )),
        before_sleep=before_sleep_log(logger, logging.WARNING)
    )
    async def call_llm_with_retry(llm, prompt: str):
        """Call LLM with intelligent retry logic."""
        try:
            response = await llm.ainvoke(prompt)
            return response
        except openai.InvalidRequestError as e:
            # Don't retry validation errors (400)
            logger.error(f"Invalid request: {e}")
            raise
        except openai.AuthenticationError as e:
            # Don't retry auth errors (401)
            logger.error(f"Auth failed: {e}")
            raise
        except Exception as e:
            logger.error(f"LLM call failed: {e}")
            raise

2.2 Circuit Breaker Pattern

from dataclasses import dataclass
from datetime import datetime, timedelta
from enum import Enum
import asyncio

class CircuitState(Enum):
    CLOSED = "closed"      # Normal operation
    OPEN = "open"          # Failing, reject requests
    HALF_OPEN = "half_open"  # Testing recovery

@dataclass
class CircuitBreakerConfig:
    failure_threshold: int = 5      # Open after N failures
    recovery_timeout: int = 60      # Try recovery after N seconds
    success_threshold: int = 2      # Close after N successes in half-open
    timeout: float = 30.0          # Request timeout

class CircuitBreaker:
    """
    Circuit breaker for external service calls.

    Prevents cascading failures by stopping calls to failing services.
    """

    def __init__(self, config: CircuitBreakerConfig):
        self.config = config
        self.failure_count = 0
        self.success_count = 0
        self.last_failure_time = None
        self.state = CircuitState.CLOSED

    def record_success(self):
        self.failure_count = 0

        if self.state == CircuitState.HALF_OPEN:
            self.success_count += 1
            if self.success_count >= self.config.success_threshold:
                self.state = CircuitState.CLOSED
                logger.info("Circuit breaker CLOSED after recovery")
        else:
            self.success_count = 0

    def record_failure(self):
        self.failure_count += 1
        self.last_failure_time = datetime.now()

        if self.failure_count >= self.config.failure_threshold:
            self.state = CircuitState.OPEN
            logger.error(
                f"Circuit breaker OPEN after {self.failure_count} failures"
            )

    def can_attempt(self) -> bool:
        """Check if we should attempt the request."""
        if self.state == CircuitState.CLOSED:
            return True

        if self.state == CircuitState.OPEN:
            # Check if recovery timeout has passed
            if (datetime.now() - self.last_failure_time) >= \
               timedelta(seconds=self.config.recovery_timeout):
                self.state = CircuitState.HALF_OPEN
                self.success_count = 0
                logger.info("Circuit breaker HALF_OPEN, testing recovery")
                return True
            return False

        return True  # HALF_OPEN

    async def call(self, func, *args, **kwargs):
        """Execute function through circuit breaker."""
        if not self.can_attempt():
            raise Exception("Circuit breaker is OPEN - service unavailable")

        try:
            # Add timeout
            result = await asyncio.wait_for(
                func(*args, **kwargs),
                timeout=self.config.timeout
            )
            self.record_success()
            return result
        except Exception as e:
            self.record_failure()
            raise e

2.3 Graceful Degradation

from typing import Optional
from enum import Enum

class DegradationLevel(Enum):
    FULL = "full"           # All features available
    DEGRADED = "degraded"   # Reduced functionality
    MINIMAL = "minimal"     # Basic features only
    OFFLINE = "offline"     # Static responses only

class DegradableService:
    """
    Service that degrades gracefully under load or failures.
    """

    def __init__(self):
        self.level = DegradationLevel.FULL

        # Initialize chains for each level
        self.full_rag_chain = self._build_rag_chain(
            retriever=vector_retriever,
            llm=ChatOpenAI(model="gpt-4")
        )

        self.degraded_chain = ChatOpenAI(model="gpt-3.5-turbo")

        self.minimal_chain = Ollama(model="llama2")  # Local

        self.offline_responses = {
            "default": "I'm experiencing technical difficulties. Please try again later."
        }

    def detect_degradation(self):
        """Automatically detect if we need to degrade."""
        metrics = get_current_metrics()

        if metrics["error_rate"] > 0.5:  # 50% errors
            self.level = DegradationLevel.OFFLINE
        elif metrics["error_rate"] > 0.2:  # 20% errors
            self.level = DegradationLevel.MINIMAL
        elif metrics["latency_p99"] > 5000:  # 5s latency
            self.level = DegradationLevel.DEGRADED
        else:
            self.level = DegradationLevel.FULL

    async def query(self, question: str) -> dict:
        """Query with appropriate degradation level."""
        self.detect_degradation()

        try:
            if self.level == DegradationLevel.FULL:
                result = await self.full_rag_chain.ainvoke(question)
                return {
                    "answer": result,
                    "level": "full",
                    "sources": result.get("source_documents", [])
                }

            elif self.level == DegradationLevel.DEGRADED:
                # Use faster, cheaper model
                result = await self.degraded_chain.ainvoke(question)
                return {
                    "answer": result,
                    "level": "degraded",
                    "warning": "Running in degraded mode - RAG disabled"
                }

            elif self.level == DegradationLevel.MINIMAL:
                # Use local model (no API dependency)
                result = await self.minimal_chain.ainvoke(question)
                return {
                    "answer": result,
                    "level": "minimal",
                    "warning": "Running in minimal mode - reduced accuracy"
                }

            else:  # OFFLINE
                return {
                    "answer": self.offline_responses["default"],
                    "level": "offline",
                    "error": "Service temporarily unavailable"
                }

        except Exception as e:
            logger.error(f"Query failed at level {self.level}: {e}")
            # Fall back to next level
            if self.level != DegradationLevel.OFFLINE:
                next_level = DegradationLevel(self.level.value + 1)
                self.level = next_level
                return await self.query(question)
            raise

2.4 Data Consistency & Backup

import asyncpg
from datetime import datetime

class ConsistentVectorStore:
    """
    Vector store with write-ahead logging for consistency.
    """

    def __init__(self, postgres_pool, vector_store):
        self.db = postgres_pool
        self.vs = vector_store

    async def add_documents_with_wal(self, documents: list):
        """
        Add documents with write-ahead logging.

        Ensures we can recover from vector store failures.
        """
        async with self.db.acquire() as conn:
            async with conn.transaction():
                # 1. Write to WAL (PostgreSQL)
                for doc in documents:
                    await conn.execute(
                        """INSERT INTO vector_wal
                           (document_id, content, metadata, timestamp)
                           VALUES ($1, $2, $3, $4)""",
                        doc.id,
                        doc.content,
                        json.dumps(doc.metadata),
                        datetime.utcnow()
                    )

                # 2. Write to vector store
                try:
                    self.vs.add_documents(documents)

                    # 3. Mark as committed in WAL
                    await conn.execute(
                        """UPDATE vector_wal
                           SET status = 'committed'
                           WHERE document_id = ANY($1)""",
                        [doc.id for doc in documents]
                    )

                except Exception as e:
                    # Rollback WAL entry
                    await conn.execute(
                        """DELETE FROM vector_wal
                           WHERE document_id = ANY($1)""",
                        [doc.id for doc in documents]
                    )
                    raise e

    async def recover_from_wal(self):
        """Recover uncommitted documents from WAL."""
        async with self.db.acquire() as conn:
            # Get uncommitted documents
            rows = await conn.fetch(
                """SELECT * FROM vector_wal
                   WHERE status = 'pending'
                   ORDER BY timestamp"""
            )

            if not rows:
                return

            logger.info(f"Recovering {len(rows)} documents from WAL")

            # Retry adding to vector store
            for row in rows:
                try:
                    doc = Document(
                        page_content=row["content"],
                        metadata=json.loads(row["metadata"])
                    )
                    self.vs.add_documents([doc])

                    # Mark as committed
                    await conn.execute(
                        "UPDATE vector_wal SET status = 'committed' WHERE id = $1",
                        row["id"]
                    )

                except Exception as e:
                    logger.error(f"Failed to recover document {row['document_id']}: {e}")

class ReplicatedVectorStore:
    """
    Multi-region vector store replication.
    """

    def __init__(self, primary_store, replica_stores: list):
        self.primary = primary_store
        self.replicas = replica_stores

    async def add_documents(self, documents: list):
        """Write to primary, replicate async."""
        # Synchronous write to primary
        self.primary.add_documents(documents)

        # Async replication (fire and forget)
        asyncio.create_task(self._replicate_to_secondaries(documents))

    async def _replicate_to_secondaries(self, documents: list):
        """Replicate to secondary stores."""
        for replica in self.replicas:
            try:
                replica.add_documents(documents)
                logger.info(f"Replicated to {replica}")
            except Exception as e:
                logger.error(f"Replication failed: {e}")

    async def query(self, query: str, k: int = 4):
        """Query with automatic failover."""
        try:
            return self.primary.similarity_search(query, k=k)
        except Exception as e:
            logger.error(f"Primary store failed: {e}")

            # Try replicas
            for replica in self.replicas:
                try:
                    logger.info(f"Failing over to replica: {replica}")
                    return replica.similarity_search(query, k=k)
                except Exception as e:
                    logger.error(f"Replica failed: {e}")

            raise Exception("All vector stores unavailable")

3.1 Multi-Region Deployment

from enum import Enum
import aiohttp

class Region(Enum):
    US_EAST_1 = "us-east-1"
    US_WEST_2 = "us-west-2"
    EU_WEST_1 = "eu-west-1"
    AP_SOUTHEAST_1 = "ap-southeast-1"

class MultiRegionLLM:
    """
    Multi-region LLM deployment with automatic failover.
    """

    def __init__(self):
        # Deploy instances in multiple regions
        self.instances = {
            Region.US_EAST_1: "https://llm-us-east.example.com",
            Region.US_WEST_2: "https://llm-us-west.example.com",
            Region.EU_WEST_1: "https://llm-eu.example.com",
        }

        self.primary = Region.US_EAST_1
        self.health_status = {region: True for region in self.instances}

    async def check_health(self, region: Region) -> bool:
        """Check if region endpoint is healthy."""
        try:
            async with aiohttp.ClientSession() as session:
                async with session.get(
                    f"{self.instances[region]}/health",
                    timeout=aiohttp.ClientTimeout(total=5)
                ) as response:
                    is_healthy = response.status == 200
                    self.health_status[region] = is_healthy
                    return is_healthy
        except Exception:
            self.health_status[region] = False
            return False

    async def get_healthy_region(self) -> Region:
        """Get the nearest healthy region."""
        # Try primary first
        if self.health_status[self.primary]:
            if await self.check_health(self.primary):
                return self.primary

        # Try other regions by proximity
        fallback_order = [
            Region.US_WEST_2,
            Region.EU_WEST_1,
            Region.AP_SOUTHEAST_1
        ]

        for region in fallback_order:
            if self.health_status[region]:
                if await self.check_health(region):
                    logger.info(f"Failing over to {region}")
                    return region

        raise Exception("No healthy regions available")

    async def invoke(self, prompt: str):
        """Invoke LLM with automatic region failover."""
        region = await self.get_healthy_region()
        url = self.instances[region]

        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(
                    f"{url}/invoke",
                    json={"prompt": prompt},
                    timeout=aiohttp.ClientTimeout(total=30)
                ) as response:
                    return await response.json()
        except Exception as e:
            # Mark region as unhealthy
            self.health_status[region] = False
            logger.error(f"Region {region} failed: {e}")

            # Retry with next healthy region
            return await self.invoke(prompt)

import asyncpg
from typing import Optional

class HAPostgresPool:
    """
    PostgreSQL connection pool with automatic failover.
    """

    def __init__(self, primary: str, standbys: list[str]):
        self.primary_url = primary
        self.standby_urls = standbys
        self.primary_pool: Optional[asyncpg.Pool] = None
        self.standby_pools: list[asyncpg.Pool] = []

    async def initialize(self):
        """Initialize connection pools."""
        # Primary pool (read-write)
        self.primary_pool = await asyncpg.create_pool(
            self.primary_url,
            min_size=5,
            max_size=20,
            command_timeout=30
        )

        # Standby pools (read-only)
        for standby_url in self.standby_standbys:
            pool = await asyncpg.create_pool(
                standby_url,
                min_size=2,
                max_size=10,
                command_timeout=30
            )
            self.standby_pools.append(pool)

    async def execute_write(self, query: str, *args):
        """Execute write query on primary."""
        if not self.primary_pool:
            raise Exception("Primary pool not initialized")

        async with self.primary_pool.acquire() as conn:
            return await conn.execute(query, *args)

    async def execute_read(self, query: str, *args):
        """Execute read query with automatic failover."""
        # Try primary first
        try:
            async with self.primary_pool.acquire() as conn:
                return await conn.fetch(query, *args)
        except Exception as e:
            logger.warning(f"Primary read failed: {e}")

        # Try standbys
        for i, pool in enumerate(self.standby_pools):
            try:
                async with pool.acquire() as conn:
                    logger.info(f"Reading from standby {i}")
                    return await conn.fetch(query, *args)
            except Exception as e:
                logger.error(f"Standby {i} failed: {e}")

        raise Exception("All databases unavailable")

class HAPinecone:

from redis import Redis
import numpy as np

class SemanticCache:
    """
    Cache LLM responses for semantically similar queries.
    """

    def __init__(self, redis_client: Redis, threshold=0.95):
        self.redis = redis_client
        self.threshold = threshold
        self.embeddings = OpenAIEmbeddings()

    async def get(self, query: str) -> Optional[str]:
        # Check for semantically similar cached query
        query_embedding = self.embeddings.embed_query(query)

        # Search cache
        cached = await self.redis.ft("idx:cache").search(
            Query(f"*=>[KNN 5 @embedding $vec AS score]")
            .return_fields("response", "score")
            .sort_by("score")
            .dialect(2),
            query_params={"vec": np.array(query_embedding).astype(np.float32).tobytes()}
        )

        if cached.docs and cached.docs[0].score >= self.threshold:
            return cached.docs[0].response

        return None

    async def set(self, query: str, response: str):
        # Store with embedding
        embedding = self.embeddings.embed_query(query)

        await self.redis.hset(
            f"cache:{hashlib.md5(query.encode()).hexdigest()}",
            mapping={
                "query": query,
                "response": response,
                "embedding": np.array(embedding).astype(np.float32).tobytes(),
                "timestamp": datetime.utcnow().isoformat()
            }
        )

class PrefetchCache:
    """
    Pre-fetch likely next queries.
    """

    def __init__(self):
        self.query_history = []
        self.graph = self._build_query_graph()

    def _build_query_graph(self):
        """
        Build a graph of likely query transitions.

        Example: After "What's the price?", users often ask "Is there a discount?"
        """
        graph = defaultdict(Counter)

        # Analyze history
        for i in range(len(self.query_history) - 1):
            current = self.query_history[i]
            next_query = self.query_history[i + 1]
            graph[current][next_query] += 1

        return graph

    async def on_query(self, query: str):
        """Called when user submits a query."""
        self.query_history.append(query)

        # Predict and pre-fetch next likely queries
        likely_next = self.graph.get(query, {})
        for next_query, score in likely_next.most_common(3):
            # Pre-fetch in background
            asyncio.create_task(self._prefetch(next_query))

# Read replicas for PostgreSQL
class ScalableDatabase:
    """
    Database with read replicas for scaling reads.
    """

    def __init__(self):
        # Single primary for writes
        self.primary = "postgresql://primary-db.example.com/db"

        # Multiple read replicas
        self.replicas = [
            "postgresql://replica-1.example.com/db",
            "postgresql://replica-2.example.com/db",
            "postgresql://replica-3.example.com/db",
        ]

        self.replica_index = 0

    async def execute_write(self, query: str, *args):
        """Write queries go to primary."""
        pool = await self.get_pool(self.primary)
        async with pool.acquire() as conn:
            return await conn.execute(query, *args)

    async def execute_read(self, query: str, *args):
        """
        Read queries are load-balanced across replicas.

        Uses round-robin with automatic failover.
        """
        # Try each replica
        for i in range(len(self.replicas)):
            replica_url = self.replicas[self.replica_index]
            self.replica_index = (self.replica_index + 1) % len(self.replicas)

            try:
                pool = await self.get_pool(replica_url)
                async with pool.acquire() as conn:
                    return await conn.fetch(query, *args)
            except Exception as e:
                logger.warning(f"Replica {replica_url} failed: {e}")
                continue

        # All replicas failed, try primary
        logger.error("All replicas failed, falling back to primary")
        return await self.execute_read_primary(query, *args)

# Vector store sharding
class ShardedVectorStore:
    """
    Partition vector store across multiple indexes.

    Enables horizontal scaling of vector search.
    """

    def __init__(self, num_shards: int):
        self.num_shards = num_shards
        self.shards = [
            pinecone.Index(f"shard-{i}")
            for i in range(num_shards)
        ]

    def _get_shard(self, document_id: str) -> int:
        """Consistent hashing to determine shard."""
        hash_value = int(hashlib.md5(document_id.encode()).hexdigest(), 16)
        return hash_value % self.num_shards

    async def upsert(self, vectors: list):
        """Upsert vectors to appropriate shards."""
        # Group vectors by shard
        shard_vectors = {i: [] for i in range(self.num_shards)}

        for vector in vectors:
            shard = self._get_shard(vector.id)
            shard_vectors[shard].append(vector)

        # Upsert to each shard in parallel
        tasks = [
            self.shards[shard].upsert(vectors)
            for shard, vectors in shard_vectors.items()
            if vectors
        ]

        await asyncio.gather(*tasks)

    async def query(self, vector: list, top_k: int = 10):
        """
        Query all shards and merge results.

        Returns top-k results across all shards.
        """
        # Query each shard
        tasks = [
            self.shards[i].query(vector=vector, top_k=top_k)
            for i in range(self.num_shards)
        ]

        results = await asyncio.gather(*tasks)

        # Merge and re-rank
        all_matches = []
        for shard_result in results:
            all_matches.extend(shard_result.matches)

        # Sort by score and return top-k
        all_matches.sort(key=lambda x: x.score, reverse=True)
        return all_matches[:top_k]

import asyncpg
from redis.asyncio import Redis as AsyncRedis

class ConnectionPoolManager:
    """
    Manages connection pools for scaling.
    """

    def __init__(self):
        self.pools = {}

    async def get_postgres_pool(self, url: str) -> asyncpg.Pool:
        """
        Get or create PostgreSQL pool.

        Pool size scales with load.
        """
        if url not in self.pools:
            pool = await asyncpg.create_pool(
                url,
                min_size=5,      # Minimum connections
                max_size=50,     # Maximum connections
                max_queries=50000,  # Recreate connection after N queries
                max_inactive_connection_lifetime=300,  # 5 minutes
                command_timeout=30
            )
            self.pools[url] = pool

        return self.pools[url]

    async def get_redis_pool(self, url: str) -> AsyncRedis:
        """Get or create Redis pool."""
        if url not in self.pools:
            pool = AsyncRedis.from_url(
                url,
                max_connections=50,
                decode_responses=True
            )
            self.pools[url] = pool

        return self.pools[url]

# Dynamic pool sizing based on load
class AutoScalingPool:
    """
    Automatically adjusts pool size based on load.
    """

    def __init__(self, base_pool):
        self.base_pool = base_pool
        self.current_size = base_pool.get_size()
        self.target_utilization = 0.7  # 70% utilization

    async def monitor_and_scale(self):
        """
        Periodically check pool utilization and adjust.

        Scale up if utilization > 70%
        Scale down if utilization < 30%
        """
        while True:
            await asyncio.sleep(60)  # Check every minute

            # Get metrics
            active_connections = self.base_pool.get_active_connections()
            total_connections = self.base_pool.get_size()
            utilization = active_connections / total_connections

            # Scale up if needed
            if utilization > self.target_utilization:
                new_size = min(total_connections * 2, 100)  # Max 100
                logger.info(f"Scaling up pool: {total_connections} → {new_size}")
                await self.base_pool.resize(new_size)

            # Scale down if underutilized
            elif utilization < 0.3 and total_connections > 10:
                new_size = max(total_connections // 2, 10)  # Min 10
                logger.info(f"Scaling down pool: {total_connections} → {new_size}")
                await self.base_pool.resize(new_size)

5.4 Load Balancing Strategies

from enum import Enum
import random

class LoadBalancingStrategy(Enum):
    ROUND_ROBIN = "round_robin"
    LEAST_CONNECTIONS = "least_connections"
    RANDOM = "random"
    IP_HASH = "ip_hash"

class LoadBalancer:
    """
    Load balancer for LLM API calls.
    """

    def __init__(self, apis: list[str], strategy: LoadBalancingStrategy):
        self.apis = apis
        self.strategy = strategy
        self.current_index = 0
        self.connections = {api: 0 for api in apis}

    def select_api(self, client_ip: str = None) -> str:
        """Select API based on strategy."""
        if self.strategy == LoadBalancingStrategy.ROUND_ROBIN:
            api = self.apis[self.current_index]
            self.current_index = (self.current_index + 1) % len(self.apis)
            return api

        elif self.strategy == LoadBalancingStrategy.LEAST_CONNECTIONS:
            # Return API with fewest active connections
            return min(self.apis, key=lambda api: self.connections[api])

        elif self.strategy == LoadBalancingStrategy.RANDOM:
            return random.choice(self.apis)

        elif self.strategy == LoadBalancingStrategy.IP_HASH:
            # Consistent hashing based on client IP
            if not client_ip:
                return random.choice(self.apis)
            hash_value = int(hashlib.md5(client_ip.encode()).hexdigest(), 16)
            index = hash_value % len(self.apis)
            return self.apis[index]

    async def call_llm(self, prompt: str, client_ip: str = None):
        """Call LLM through load balancer."""
        api = self.select_api(client_ip)

        self.connections[api] += 1
        try:
            response = await call_openai_api(api, prompt)
            return response
        finally:
            self.connections[api] -= 1

async def metrics_middleware(request, call_next):
    start_time = time.time()

    # Process request
    response = await call_next(request)

    # Record metrics
    duration = time.time() - start_time

    request_counter.labels(
        endpoint=request.url.path,
        method=request.method,
        status=response.status_code

    request_duration.labels(
        endpoint=request.url.path,
        method=request.method
